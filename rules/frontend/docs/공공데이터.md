const BASE = 'https://apis.data.go.kr/B553881/Parking';
const key = encodeURIComponent(process.env.SERVICE_KEY);

async function fetchJSON(path, qs) {
  const url = `${BASE}/${path}?serviceKey=${key}&format=2&numOfRows=100&pageNo=1${qs||''}`;
  const res = await fetch(url);
  if (!res.ok) throw new Error(`API ${path} ${res.status}`);
  return res.json();
}
